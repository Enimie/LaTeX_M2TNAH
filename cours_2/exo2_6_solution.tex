\newcommand{\prg}[1]{\S #1\xspace}%il y a un argument (le numéro de paragraphe); \xspace permet d'obtenir le bon espace selon ce qui suit

%la commande \oeuvre
\newcommand{\oeuvre}[3]{#1, \emph{#2} (#3)}%Trois arguments, l'auteur, le titre et la date. Le titre (#2) est en emphase; la date (#3) est entre parenthèses

\newcommand{\siecle}[1]{\textsc{\romannum{#1}}\textsuperscript{e} siècle\xspace}

%\newenvironment{chronologie}[1]{\begin{framed}\begin{center}\textbf{#1}\end{center}\begin{description}}{\end{description}\end{framed}}

%l'environnement chronologie
\newenvironment{chronologie}[1]{\par #1\begin{description}}{\end{description}}
%un argument (le titre de la chronologie); après le titre, nouveau paragraphe et environnement description, qui permet d'associer un élément (l'argument optionnel de  \item) à sa description. L'argument de \item est mis par défaut en gras dans l'environnement description.
